S     = 'typeInt' SR (OK) 
      | NotIntType Decl S
      | 'typeVoid' DeclFun S
      | EPSILON


SR  =  'funcDecl' SAux (OK) 
    | DeclArrOpc 'funcDecl' 'identifier' DeclFun S
    | DeclArrOpc 'identifier' DeclVar S
    | 'identifier' DeclVar S

SAux = 'main' Main (OK) 
     | 'identifier' DeclFun S (OK) 

NotIntType  = 'typeFloat' (OK)
            | 'typeString' (OK)
            | 'typeChar' (OK)
            | 'typeBool' (OK)

Decl = DeclArrOpc Declx (OK)

Declx = 'funcDecl' 'identifier' DeclFun (OK)
      | 'identifier' DeclVar (OK)

DeclFun = '(' Params ')' Body (OK)  

VarType = 'typeInt' (OK)
        | 'typeFloat' (OK)
        | 'typeString' (OK)
        | 'typeChar' (OK)
        | 'typeBool' (OK)

DeclArrOpc = '[' ArrSizeOpc ']' (OK)
           | EPSILON

ArrSizeOpc = ArrSizeObg  (OK)
           | EPSILON (OK)

ArrSizeObg = 'identifier' (OK)
            | 'intVal'     (OK)        

Params = VarType 'identifier' ArrOpc Paramsx (OK)
            | EPSILON (OK)

ArrOpc = '[' ArrSizeObg ']' (OK)
        | EPSILON (OK)

Paramsx = ',' VarType 'identifier' ArrOpc Paramsx (OK)
        | EPSILON (OK)

Body = '{' Content '}'  (OK)

Content = Command Content  (OK)
        | DeclVar Content
        | 'identifier' '(' ParamsCall ')' ';' Content (OK)
        | 'funcRtn' Rtn ';' (OK)
        | LAttr ';' Content
        | EPSILON

Command = 'cmdIf' '(' Eb ')' Body LElif CmdElse (OK)
        | 'cmdWhile' '(' Eb ')' Body (OK)
        | 'cmdFor' 'identifier' '=' '(' Ea ',' Ea ',' Ea ')' Body 
        | 'funcRead' '(' LIdentifier ')' ';' (OK)
        | 'funcPrint' '(' 'stringVal' PrintParams ')' ';' (OK)

CmdElse = 'cmdElse' Body (OK)
        | EPSILON (OK)

LElif   = 'cmdElif' '(' Eb ')' Body LElif (OK)
        | EPSILON (OK)

LIdentifier = 'identifier' LIdentifierx (OK)
            | EPSILON (OK)

LIdentifierx = ',' 'identifier' LIdentifierx (OK)
            | EPSILON (OK)
 
PrintParams = ',' Eb PrintParams  (OK)
            | EPSILON (OK)

DeclVar = LVar ';'  
               
LVar    = Varx LVarx
        | EPSILON

LVarx = ',' Var LVarx
    |EPSILON

Var = 'identifier' Varx
 
Varx = '=' Ec
     | DeclArrObg
     | EPSILON

ParamsCall  = Ec ParamsCallx (OK)
            | EPSILON (OK)

ParamsCallx = ',' Ec ParamsCallx (OK)
            |EPSILON (OK)

Rtn    = Ec   (OK)
       | EPSILON (Caso void)

LAttr = 'identifier' '=' Ec LAttrx
      | 'identifier' '[' Ea ']' '=' Ec LAttrx

LAttrx = ',' identifier '=' Ec LAttrx
        | ',' '[' Ea ']' '=' Ec LAttrx
        |EPSILON

Main = '(' ')' Body  (OK)

Ec          = Eb Ecx

Ecx = 'opConcat' Eb Ecx
    | EPSILON

Eb          = Tb Ebx

Ebx = 'opOr' Tb Ebx
    | EPSILON

Tb         = Fb Tbx

Tbx = 'opAnd' Fb Tbx
    | EPSILON

Fb  = 'opNot' Fb
    | Ra Fbx

Fbx = 'opGtrThan' Fb Fbx
    | 'opLessThan' Fb Fbx
    | 'opGtrEqual' Fb Fbx
    | 'opLessEq' Fb Fbx
    | EPSILON

Ra          = Ea Rax

Rax = 'opEquals' Ea Rax
    | 'opDiff' Ea Rax
    | EPSILON              

Ea          = Ta Eax

Eax = 'opAdd' Ta Eax
    | 'opSub' Ta Eax
    | EPSILON

Ta          = Fa Tax

Tax = 'opMult' Fa Tax
    | 'opDiv' Fa Tax
    | 'opMod' Fa Tax
    | EPSILON               

Fa          = '(' Ec ')'
            | 'opSub' Fa        
            | VarOrFunc
            | 'intVal'
            | 'floatVal'
            | 'stringVal'
            | 'charVal'
            | 'boolVal'


VarOrFunc   = 'identifier' VarOrFuncx (OK) 

VarOrFuncx  = '[' Ea ']' (OK) 
            | '(' ParamsCall ')' (OK) 
            |EPSILON (OK)
